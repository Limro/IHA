\section*{Problem 8}
$S=\{2, 4, 5, 9\}, T=\{1,3,5,7,9\}, V=\{4, 5, 6\}$
\begin{enumerate}[a]
\item $|S\cap T| = \{5, 9\} =2$
\item $|T\cup V| = \{1, 3, 4, 5 ,6, 7, 9\} =7$
\item $|V-S| = \{2, 1, 1 ,-9\} = 3$
\item $|S-V| = \{-2, -1, -1, 9\} = 3$
\item $|T\cap S \cup V| = \{5,9\} \cup V = \{4, 5, 6, 9\} = 4$
\item $|S \cup T \cup V| = \{1,2,3,4,5,6,7,9\} = 8$
\end{enumerate}
