\begin{align}
x &= [x_1, x_2 \ldots, x_m] \\
\sum x_i &= 1\\
y &= [y_1, y_2 \ldots, y_n] \\
\sum y_i &= 1\\
E[x,y] &= \sum_{j=1}^n \sum_{i=1}^m x_i \cdot a_{ij} \cdot y_i \\
	&= x^T \cdot A \cdot y
\end{align}
